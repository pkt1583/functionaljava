003    import fj.data.List;

006    import java.lang.ref.SoftReference;

008    /**
009     * A product-1. Also, the identity monad.
010     *
011     * @version %build.number%<br>
012     *          <ul>
013     *          <li>$LastChangedRevision$</li>
014     *          <li>$LastChangedDate$</li>
015     *          </ul>
016     */
017    public abstract class P1<A> {
018      /**
019       * Access the first element of the product.
020       *
021       * @return The first element of the product.
022       */
023      public abstract A _1();
024    
025      /**
026       * Map the element of the product.
027       *
028       * @param f The function to map with.
029       * @return A product with the given function applied.
030       */
031      public <X> P1<X> map(final F<A, X> f) {
032        return new P1<X>() {
033          public X _1() {
034            return f.f(P1.this._1());
035          }
036        };
037      }
038    
039      /**
040       * Returns a function that returns the first element of a product.
041       *
042       * @return A function that returns the first element of a product.
043       */
044      public static <A> F<P1<A>, A> __1() {
045        return new F<P1<A>, A>() {
046          public A f(final P1<A> p) {
047            return p._1();
048          }
049        };
050      }
051    
052      /**
053       * Promote any function to a transformation between P1s.
054       *
055       * @param f A function to promote to a transformation between P1s.
056       * @return A function promoted to operate on P1s.
057       */
058      public static <A, B> F<P1<A>, P1<B>> fmap(final F<A, B> f) {
059        return new F<P1<A>, P1<B>>() {
060          public P1<B> f(final P1<A> a) {
061            return a.map(f);
062          }
063        };
064      }
065    
066      /**
067       * Binds the given function to the value in a product-1 with a final join.
068       *
069       * @param a A value in a product-1 to which to apply a function.
070       * @param f A function to apply to the value in a product-1.
071       * @return The result of applying the given function to the value of given product-1.
072       */
073      public static <A, B> P1<B> bind(final P1<A> a, final F<A, P1<B>> f) {
074        return new P1<B>() {
075          public B _1() {
076            return f.f(a._1())._1();
077          }
078        };
079      }
080    
081      /**
082       * Promotes the given function so that it returns its value in a P1.
083       *
084       * @param f A function to have its result wrapped in a P1.
085       * @return A function whose result is wrapped in a P1.
086       */
087      public static <A, B> F<A, P1<B>> curry(final F<A, B> f) {
088        return new F<A, P1<B>>() {
089          public P1<B> f(final A a) {
090            return new P1<B>() {
091              public B _1() {
092                return f.f(a);
093              }
094            };
095          }
096        };
097      }
098    
099      /**
100       * Performs function application within a P1 (applicative functor pattern).
101       *
102       * @param ca The P1 to which to apply a function.
103       * @param cf The P1 function to apply.
104       * @return A new P1 after applying the given P1 function to the first argument.
105       */
106      public static <A, B> P1<B> apply(final P1<A> ca, final P1<F<A, B>> cf) {
107        return bind(cf, new F<F<A, B>, P1<B>>() {
108          public P1<B> f(final F<A, B> f) {
109            return fmap(f).f(ca);
110          }
111        });
112      }
113    
114      /**
115       * Binds the given function to the values in the given P1s with a final join.
116       *
117       * @param ca A given P1 to bind the given function with.
118       * @param cb A given P1 to bind the given function with.
119       * @param f  The function to apply to the values in the given P1s.
120       * @return A new P1 after performing the map, then final join.
121       */
122      public static <A, B, C> P1<C> bind(final P1<A> ca, final P1<B> cb, final F<A, F<B, C>> f) {
123        return apply(cb, fmap(f).f(ca));
124      }
125    
126      /**
127       * Joins a P1 of a P1 with a bind operation.
128       *
129       * @param a The P1 of a P1 to join.
130       * @return A new P1 that is the join of the given P1.
131       */
132      public static <A> P1<A> join(final P1<P1<A>> a) {
133        return bind(a, Function.<P1<A>>identity());
134      }
135    
136      /**
137       * Promotes a function of arity-2 to a function on P1s.
138       *
139       * @param f The function to promote.
140       * @return A function of arity-2 promoted to map over P1s.
141       */
142      public static <A, B, C> F<P1<A>, F<P1<B>, P1<C>>> liftM2(final F<A, F<B, C>> f) {
143        return Function.curry(new F2<P1<A>, P1<B>, P1<C>>() {
144          public P1<C> f(final P1<A> pa, final P1<B> pb) {
145            return bind(pa, pb, f);
146          }
147        });
148      }
149    
150      /**
151       * Turns a List of P1s into a single P1 of a List.
152       *
153       * @param as The list of P1s to transform.
154       * @return A single P1 for the given List.
155       */
156      public static <A> P1<List<A>> sequence(final List<P1<A>> as) {
157        return as.foldRight(liftM2(List.<A>cons()), P.p(List.<A>nil()));
158      }
159    
160      /**
161       * A first-class version of the sequence method for lists of P1s.
162       *
163       * @return A function from a List of P1s to a single P1 of a List.
164       */
165      public static <A> F<List<P1<A>>, P1<List<A>>> sequenceList() {
166        return new F<List<P1<A>>, P1<List<A>>>() {
167          public P1<List<A>> f(final List<P1<A>> as) {
168            return sequence(as);
169          }
170        };
171      }
172    
173      /**
174       * Turns a stream of P1s into a single P1 of a stream.
175       *
176       * @param as The stream of P1s to transform.
177       * @return A single P1 for the given stream.
178       */
179      public static <A> P1<Stream<A>> sequence(final Stream<P1<A>> as) {
180        return as.foldRight(liftM2(Stream.<A>cons()), P.p(Stream.<A>nil()));
181      }
182    
183      /**
184       * Turns an array of P1s into a single P1 of an array.
185       *
186       * @param as The array of P1s to transform.
187       * @return A single P1 for the given array.
188       */
189      public static <A> P1<Array<A>> sequence(final Array<P1<A>> as) {
190        return new P1<Array<A>>() {
191          public Array<A> _1() {
192            return as.map(P1.<A>__1());
193          }
194        };
195      }
196    
197      /**
198       * Provides a memoising P1 that remembers its value.
199       *
200       * @return A P1 that calls this P1 once and remembers the value for subsequent calls.
201       */
202      public P1<A> memo() {
203        final P1<A> self = this;
204        return new P1<A>() {
205          private final Object latch = new Object();
206          @SuppressWarnings({"InstanceVariableMayNotBeInitialized"})
207          private volatile SoftReference<A> v;
208    
209          public A _1() {
210            A a = v != null ? v.get() : null;
211            if (a == null)
212              synchronized (latch) {
213                if (v == null || v.get() == null)
214                  a = self._1();
215                v = new SoftReference<A>(a);
216              }
217            return a;
218          }
219        };
220      }
221    
222      /**
223       * Returns a constant function that always uses this value.
224       *
225       * @return A constant function that always uses this value. 
226       */
227      public <B> F<B, A> constant() {
228        return new F<B, A>() {
229          public A f(final B b) {
230              return _1();
231          }
232        };
233      }
